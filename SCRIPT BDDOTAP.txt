CREATE DATABASE BDDOTAP;
USE BDDOTAP;

CREATE TABLE TIPO_USUARIO
(
	ID_TIPOUSU INT NOT NULL PRIMARY KEY,
	NOMBRE_TIPOUSU VARCHAR(50)
);

CREATE TABLE USUARIO
(
	ID_USU INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	NOMBRE_USU varchar(100) UNIQUE KEY,
	CLAVE_USU varchar(50) NOT NULL,
	CORREO_USU varchar(50) NOT NULL UNIQUE KEY,
        TELEFONO_USU VARCHAR(9),
	ID_TIPOUSU INT NOT NULL,
	ESTADO CHAR(1),
	FOREIGN KEY(ID_TIPOUSU)	REFERENCES TIPO_USUARIO(ID_TIPOUSU)
);
